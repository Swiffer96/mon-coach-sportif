// Afficher un onglet

function showTab(tabName) {

    document.querySelectorAll('.tab-content').forEach(tab => tab.classList.remove('active'));

    document.querySelectorAll('.nav-tab').forEach(tab => tab.classList.remove('active'));

    document.getElementById(tabName).classList.add('active');

    event.target.classList.add('active');


    if (tabName === 'suivi') updateHistory();

    if (tabName === 'programme') updateWeekProgram();

}


// Mettre à jour les records

function updateRecords() {

    for (const [exercise, weight] of Object.entries(records)) {

        const lastSession = Object.values(sessions).find(session =>

            session.exercises && session.exercises[exercise]

        );


        const weightDisplay = lastSession && lastSession.exercises[exercise] && lastSession.exercises[exercise].weight

            ? `${lastSession.exercises[exercise].weight}kg x ${lastSession.exercises[exercise].reps || 'N/A'} reps`

            : 'Aucune donnée';


        recordsContainer.innerHTML += `

            <li class="exercise-item">

                <div class="exercise-info">

                    <h3>${exercise}</h3>

                    <p>Dernière séance: ${weightDisplay}</p>

                </div>

                <span class="record">Record: ${weight}kg</span>

            </li>

        `;

    }

}

// Enregistrer une séance

function saveSession(day) {

    const exercises = {};

    const inputs = document.querySelectorAll(`[data-day="${day}"]`);

    const notes = document.getElementById(`notes-${day}`).value;


    inputs.forEach(input => {

        const exercise = input.dataset.exercise;

        const type = input.placeholder.includes('Poids') ? 'weight' :

                     input.placeholder.includes('Reps') ? 'reps' : 'rpe';


        if (!exercises[exercise]) exercises[exercise] = { weight: null, reps: null, rpe: null };

        exercises[exercise][type] = input.value;

    });


    sessions[day] = {

        date: new Date().toISOString().split('T')[0],

        time: new Date().toLocaleTimeString(),

        exercises: exercises,

        notes: notes

    };


    localStorage.setItem('sessions', JSON.stringify(sessions));


    // Mettre à jour les records

    for (const [exercise, data] of Object.entries(exercises)) {

        if (data.weight && !isNaN(data.weight)) {

            const weight = parseFloat(data.weight);

            if (!records[exercise] || weight > records[exercise]) {

                records[exercise] = weight;

            }

        }

    }

    localStorage.setItem('records', JSON.stringify(records));


    updateRecords();

    updateHistory();

    renderCharts();


    alert(`Séance du ${day} enregistrée avec succès !`);

}


// Commencer une séance (enregistre l'heure de début)

function startSession(day) {

    if (!sessions[day]) sessions[day] = {};

    sessions[day].startTime = new Date().toISOString();

    localStorage.setItem('sessions', JSON.stringify(sessions));

    alert(`Séance du ${day} commencée à ${new Date().toLocaleTimeString()}`);

}

// Gestion des défis

function loadChallenge() {

    if (!currentChallenge) {

        // Sélectionner un défi aléatoire

        const randomIndex = Math.floor(Math.random() * challenges.length);

        currentChallenge = challenges[randomIndex];

        currentChallenge.startDate = new Date().toISOString();

        currentChallenge.progress = 0;

        localStorage.setItem('currentChallenge', JSON.stringify(currentChallenge));

    }


    document.getElementById('challenge-text').textContent = currentChallenge.name;

    document.getElementById('challenge-progress').textContent = `${Math.round((currentChallenge.progress / currentChallenge.target) * 100)}%`;

    document.getElementById('challenge-bar').style.width = `${(currentChallenge.progress / currentChallenge.target) * 100}%`;

}


function startChallenge() {

    if (!currentChallenge) loadChallenge();

    alert(`Défi "${currentChallenge.name}" commencé !`);

}


function completeChallenge() {

    if (currentChallenge.progress >= currentChallenge.target) {

        alert(`Félicitations ! Tu as terminé le défi "${currentChallenge.name}" !`);

        currentChallenge = null;

        localStorage.removeItem('currentChallenge');

        loadChallenge();

    } else {

        alert(`Tu as progressé à ${Math.round((currentChallenge.progress / currentChallenge.target) * 100)}%. Continue !`);

    }

}


// Minuteur

let timer;

let timeLeft = 90; // 1min30 par défaut


function startTimer() {

    const minutes = parseInt(document.getElementById('timer-minutes').value) || 1;

    const seconds = parseInt(document.getElementById('timer-seconds').value) || 30;

    timeLeft = minutes * 60 + seconds;


    timer = setInterval(() => {

        if (timeLeft <= 0) {

            clearInterval(timer);

            document.getElementById('timer-display').textContent = "Temps écoulé !";

            alert("Temps de repos terminé !");

            return;

        }

        timeLeft--;

        const mins = Math.floor(timeLeft / 60);

        const secs = timeLeft % 60;

        document.getElementById('timer-display').textContent = `${mins}:${secs < 10 ? '0' : ''}${secs}`;

    }, 1000);

}


function stopTimer() {

    clearInterval(timer);

}


function resetTimer() {

    clearInterval(timer);

    const minutes = parseInt(document.getElementById('timer-minutes').value) || 1;

    const seconds = parseInt(document.getElementById('timer-seconds').value) || 30;

    timeLeft = minutes * 60 + seconds;

    document.getElementById('timer-display').textContent = `${minutes}:${seconds < 10 ? '0' : ''}${seconds}`;

}


function closeTimer() {

    document.getElementById('timer-modal').style.display = 'none';

    resetTimer();

}

// Exporter en CSV

function exportToCSV() {

    let csv = "Date,Séance,Exercice,Poids (kg),Répétitions,RPE,Notes\n";


    for (const [day, session] of Object.entries(sessions)) {

        for (const [exercise, data] of Object.entries(session.exercises)) {

            csv += `"${session.date}","${day}","${exercise}","${data.weight || ''}","${data.reps || ''}","${data.rpe || ''}","${session.notes || ''}"\n`;

        }

    }


    const blob = new Blob([csv], { type: 'text/csv;charset=utf-8;' });

    const link = document.createElement('a');

    const url = URL.createObjectURL(blob);

    link.setAttribute('href', url);

    link.setAttribute('download', 'historique-seances.csv');

    link.style.visibility = 'hidden';

    document.body.appendChild(link);

    link.click();

    document.body.removeChild(link);

}
